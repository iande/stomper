def transmit_frame(frame)
  write([ frame.command, Stomper::Frames::LINE_DELIMITER,
    serialize_headers(frame.headers), Stomper::Frames::LINE_DELIMITER,
    frame.body, Stomper::Frames::TERMINATOR.chr].join)
end
